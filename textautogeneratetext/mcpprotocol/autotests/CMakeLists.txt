macro(add_textautogenerate_mcpprotocol_unittest _source)
    get_filename_component(_name ${_source} NAME_WE)
    ecm_add_test(${_source} ${_name}.h TEST_NAME ${_name}
        NAME_PREFIX "textautogenerate-mcpprotocol-"
        LINK_LIBRARIES Qt::Test KF6::TextAutoGenerateTextMcpProtocol
    )
endmacro()
